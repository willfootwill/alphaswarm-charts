1. Simple Example - Getting Started

A basic footswarm chart showing three groups with different distributions. Perfect for understanding the fundamentals.

Example Code:

import { createFootswarmChart } from './src/footswarm.js';
import { simpleData } from './data/sample-datasets.js';

const chart = createFootswarmChart(simpleData, {
    x: 'value',
    y: 'category',
    xLabel: 'Value',
    yLabel: 'Groups',
    opacity: 0.6,
    jitter: 0.5,
    showMean: true,
    showMedian: true
});

document.getElementById('chart-container').appendChild(chart);
